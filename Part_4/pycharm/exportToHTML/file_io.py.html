47   ## Another kind of interaction of the code with 
48   ## the outside world is through files. 
49    
50   ## The simplest and easiest kind of data to work with is 
51   ## accessing files on the same computer as your script. 
52   ## Python has a built-in way to open files, the 
53   ## open() function. 
54    
55   ## open() returns a "file object", which is basically 
56   ## a thing that knows how to read and write the file 
57   ## whose name you pass into the open() function. 
58    
59   ## fp = open(filename,mode) 
60    
61   ## The first argument to open() is the path to the file 
62   ## to open, and the second argument is the "mode" which 
63   ## identifies your intention to open the file. The mode 
64   ## is usually one of the following: 
65   ##    'r' for read-only 
66   ##    'w' to write to a file by overwriting 
67   ##    'a' to write to a file by appending at the end 
68   ##    'r+' to read and write 
69    
70   ## A file object has methods. `write()` takes a 
71   ## message and writes to the file object. `close()` 
72   ## method closes the pipe to the file. 
73    
74   msg1='First line of file' 
75    
76   fp = open('log.txt','w') 
77   fp.write(msg1) 
78   fp.write('\n') 
79    
80   # fp.write(msg1,'\n') # Show fileobject.write() takes exactly one argument 
81   # fp.write(msg1+'\n') # but this is acceptable 
82    
83   msg2='Second line of file' 
84   fp.write(msg2) 
85   fp.write('\n') 
86    
87   print(fp.closed) # We are querying if the file is closed 
88    
89   if (not fp.closed): 
90       fp.close() # Don't forget to close your files when you are done 
91    
92   print(fp.closed) # We are querying if the file is closed 
93

113  ## Let's look at some other methods of 
114  ## file objects 
115   
116  ## `read()` returns the entire contents of the file 
117   
118  ## `readline()` returns one line at a time from the file. 
119   
120  ## `tell()` returns the current position of the file 
121  ## read/write pointer within the file 
122   
123  ## `seek()` changes the position in the file object. 
124  ## fp.seek(offset, from_what) : from_what indicates 
125  ## the reference point, offset is the position to be 
126  ## moved reference to from_what. from_what can be 0 
127  ## (default, beginning of the file), 1 (current position) 
128  ## and 2 (end of file) 
129   
130   
131  fp = open('log.txt','r') 
132   
133  print(fp.read(1)) 
134  print(fp.read(1)) 
135  print('Current position in the file:', fp.tell()) 
136  fp.seek(1 , 0) 
137  print('Current position in the file:', fp.tell()) 
138  print(fp.readline()) 
139   
140  print(fp.read()) 
141   
142  fp.closed 
143   
144  if (not fp.closed): # We are querying if the file is closed 
145      fp.close 
146   
147  ## Closing file: 
148  # - Avoids using unnecessary system resources 
149  # in the form of file descriptors on handles 
150  # - Avoids race conditions and code failures 
151  # depending on the access rights

154  ## closing the file use `with open() as`
